# ca organization
genrsa -out ca-key.pem -des 2048
req -new -key ca-key.pem -out ca-csr.pem
x509 -req -in ca-csr.pem -signkey ca-key.pem -out ca-cert.pem

# server ca
genrsa -out server-key.pem 2048
req -new -key server-key.pem -config <PATH_TO_OPENSSL.CNF> -out server-csr.pem
x509 -req -CA ca-cert.pem -CAkey ca-key.pem -CAcreateserial -in server-csr.pem -out server-cert.pem -extensions v3_req
